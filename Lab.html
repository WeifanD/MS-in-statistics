# NRstep method
NRstep <- function(y,theta){
  n <- length(y)
  U <- (-2*n/theta)+(2*sum(y^2)/theta^3)
  Uprime <- (2*n/theta^2)-(6*sum(y^2)/theta^4)
  thetanew <- theta - U/Uprime
  list(U=U,Uprime=Uprime,theta=theta,thetanew=thetanew)
}

# scoring method
Scoring <- function(y,theta){
  n <- length(y)
  U <- -2*n/theta+2*sum(y^2)/theta^3
  I <- 4*n/theta^2
  thetanew <- theta - U/I
  list(U=U,I=I,theta=theta,thetanew=thetanew)
}

2. Poisson regression

# Poisson regression
MLE <- function(y,beta1,beta2,x){
  mu <- beta1 + beta2*x
  X <- cbind(1,x)
  W <- diag(1/mu)
  beta <- solve(t(X)%*%W%*%X,t(X)%*%W%*%y)
  beta
}

y <- c(2,3,6,7,8,9,10,12,15)
x <- c(-1,-1,0,0,0,0,1,1,1)

# Obviously doing the iterations by hand gets tedious very quickly.
# Better write an R function to fit this GLM:

irwls.pois <- function(y, X, beta0, tol=0.0001, maxit=100)
{
  # y: y vector
  # beta0: vector with initial values of parameter estimates
  beta<- beta0
  counter <- 0
  for (i in 1:maxit) {
    beta.old <- beta            # store old value of beta                 
    # Design matrix X
    # Xbeta:
    eta <- X%*% beta
    mu <- exp(eta) 
    # W matrix:
    W <- diag(as.vector(mu))
    # z + eta vector:
    z <- y/mu-1
    # New parameter vector:
    beta <- solve(t(X)%*% W%*% X)%*%t(X)%*% W%*% (eta+z)
    counter <- counter+1
    print(c(counter, beta))
    if (sum(abs((beta-beta.old)/beta)<tol)) break
  }
  return(list(beta=beta, iter=counter))
}
